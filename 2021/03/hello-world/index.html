/// <summary>
        /// 取得所有台股名稱列表
        /// </summary>
        /// <returns></returns>
        public async Task<CategoryViewModel> GetTaiwanStockInfos()
        {

            using (var httpClientHandler = new HttpClientHandler { CookieContainer = _cookieContainer })
            {
                var _httpClient = new HttpClient(httpClientHandler);

                Uri uri = new Uri($"{AppSettingsConstants.WebAPIUrl}api/Category/StockInfos");

                HttpResponseMessage response = await _httpClient.GetAsync(uri).ConfigureAwait(false);

                if (response.IsSuccessStatusCode)
                {
                    string userJson = await response.Content.ReadAsStringAsync();
                    if (!string.IsNullOrEmpty(userJson))
                    {
                        var categoryStockInfos = JsonConvert.DeserializeObject<List<CategoryViewModel>>(userJson);
                        var categoryStockInfo = categoryStockInfos.FirstOrDefault();
                        return categoryStockInfo;
                    }
                }
            }

            return null;
        }
